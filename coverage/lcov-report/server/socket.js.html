import socketIO from 'socket.io';
 
import IoGame from './models/IoGame';
import ev from '../shared/events';
import actions from './actions';
 
const redTetris = (server) => {
  const io = socketIO(server, {
    pingInterval: 5000,
    pingTimeout: 15000,
  });
 
  const redGame = new IoGame(io);
 
  redGame.io.on('connect', (socket) => {
    /* Socket events */
    actions.socketConnect(socket, redGame);
    socket.on(ev.CLIENT_DISCONNECT, () => actions.socketDisconnect(socket, redGame));
    socket.on(ev.ERROR, () => actions.socketError(socket, redGame));
 
    /* Player events */
    socket.on(ev.req_LOGIN, (data, callback) => actions.resLogin(socket, data, redGame, callback));
    socket.on(ev.req_LOGOUT, (data) => actions.resLogout(socket, data, redGame));
 
    socket.on(ev.req_PLAYER_LOOSE, (data) => actions.resLoose(socket, data, redGame));
 
    /* Game events */
    socket.on(ev.req_START_GAME, (data) => actions.resStartGame(socket, data, redGame));
    socket.on(ev.req_UPDATE_GAME_CHAT, (data) => actions.resChat(socket, data, redGame));
    socket.on(ev.req_UPDATE_GAME_OWNER, (data) => actions.resOwner(socket, data, redGame));
    socket.on(ev.req_PIECES, (data) => actions.resPieces(socket, data, redGame));
    socket.on(ev.req_COLLISION, (data) => actions.resUpdateCollision(socket, data, redGame));
    socket.on(ev.req_GAME_PLAYER, (data) => actions.resGamePlayer(socket, data, redGame));
    socket.on(ev.req_MALLUS, (data) => actions.resMallus(socket, data, redGame));
  });
};
 
module.exports = redTetris;
